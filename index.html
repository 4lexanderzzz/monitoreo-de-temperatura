  document.getElementById("lastTs").textContent =
    `${d.ts.toLocaleString()} ms`;

  // Agregar al historial local si no existe
  if (!historial.find(r => r.ts === d.ts)) {
    historial.unshift({ ts: d.ts, entrada: d.entrada, salida: d.salida, diferencia: d.diferencia });
    if (historial.length > 200) historial.pop();
    actualizarTabla();
    actualizarGrafica();
  }

  midiendo = snap.val() === true;
  sincronizarUI();

  const data = snap.val();

  historial = Object.values(data).reverse();
  actualizarTabla();
  actualizarGrafica();

function sincronizarUI() {
  const badge = document.getElementById("statusBadge");
  const texto = document.getElementById("statusText");
  const btnStart = document.getElementById("btnStart");
  const btnStop  = document.getElementById("btnStop");

  }
}

function actualizarTabla() {

// ══════════════════════════════════════════════
//  GRÁFICA SVG
// ══════════════════════════════════════════════
function actualizarGrafica() {
  const datos = historial.slice(0, 30).reverse();
  if (datos.length < 2) return;

  const svg   = document.getElementById("chartSvg");
  const W = 900, H = 180, PAD = 16;

  // Rango de temperatura
  const allTemps = datos.flatMap(d => [d.entrada, d.salida]);
  const minT = Math.min(...allTemps) - 1;
  const maxT = Math.max(...allTemps) + 1;

  const xOf = i => PAD + (i / (datos.length - 1)) * (W - 2 * PAD);
  const yOf = t => PAD + (1 - (t - minT) / (maxT - minT)) * (H - 2 * PAD);

  const polylinePoints = (field) =>
    datos.map((d, i) => `${xOf(i)},${yOf(d[field])}`).join(" ");

  const areaPath = (field, color) => {
    const pts = datos.map((d, i) => `${xOf(i)},${yOf(d[field])}`).join(" ");
    const first = `${xOf(0)},${H}`;
    const last  = `${xOf(datos.length - 1)},${H}`;
    return `M ${first} L ${pts} L ${last} Z`;
  };

  svg.innerHTML = `
    <defs>
      <linearGradient id="gradEntrada" x1="0" y1="0" x2="0" y2="1">
        <stop offset="0%" stop-color="#38bdf8" stop-opacity="0.5"/>
        <stop offset="100%" stop-color="#38bdf8" stop-opacity="0"/>
      </linearGradient>
      <linearGradient id="gradSalida" x1="0" y1="0" x2="0" y2="1">
        <stop offset="0%" stop-color="#f97316" stop-opacity="0.5"/>
        <stop offset="100%" stop-color="#f97316" stop-opacity="0"/>
      </linearGradient>
    </defs>

    <!-- Áreas rellenas -->
    <path d="${areaPath('entrada')}" fill="url(#gradEntrada)"/>
    <path d="${areaPath('salida')}"  fill="url(#gradSalida)"/>

    <!-- Líneas -->
    <polyline points="${polylinePoints('entrada')}"
      fill="none" stroke="#38bdf8" stroke-width="2.2"
      stroke-linecap="round" stroke-linejoin="round"/>
    <polyline points="${polylinePoints('salida')}"
      fill="none" stroke="#f97316" stroke-width="2.2"
      stroke-linecap="round" stroke-linejoin="round"/>

    <!-- Puntos último dato -->
    <circle cx="${xOf(datos.length-1)}" cy="${yOf(datos[datos.length-1].entrada)}"
      r="4" fill="#38bdf8"/>
    <circle cx="${xOf(datos.length-1)}" cy="${yOf(datos[datos.length-1].salida)}"
      r="4" fill="#f97316"/>

    <!-- Leyenda -->
    <rect x="10" y="8" width="12" height="3" rx="1.5" fill="#38bdf8"/>
    <text x="26" y="12" fill="#7d8590" font-size="10">Entrada</text>
    <rect x="72" y="8" width="12" height="3" rx="1.5" fill="#f97316"/>
    <text x="88" y="12" fill="#7d8590" font-size="10">Salida</text>
  `;
}

// ══════════════════════════════════════════════
//  EXPORTAR EXCEL
// ══════════════════════════════════════════════
function exportarExcel() {
  if (historial.length === 0) { alert("No hay datos para exportar."); return; }

  const filas = historial.map((r, i) => ({
    "N°":              i + 1,
    "Tiempo (ms)":     r.ts,
    "Entrada (°C)":    r.entrada,
    "Salida (°C)":     r.salida,
    "Diferencia (°C)": r.diferencia
  }));

  const ws = XLSX.utils.json_to_sheet(filas);

  // Ancho de columnas
  ws['!cols'] = [
    { wch: 6 }, { wch: 14 }, { wch: 14 }, { wch: 13 }, { wch: 16 }
  ];

  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws, "Mediciones");
  XLSX.writeFile(wb, `temperaturas_${Date.now()}.xlsx`);
}

function exportarTxt() {
  if (historial.length === 0) { alert("No hay datos para exportar."); return; }

  const cabecera = "N°\tTiempo(ms)\tEntrada(°C)\tSalida(°C)\tDiferencia(°C)";
  const filas = historial.map((r, i) =>
    `${i+1}\t${r.ts}\t${r.entrada.toFixed(2)}\t${r.salida.toFixed(2)}\t${r.diferencia.toFixed(2)}`
  );

  const contenido = [cabecera, ...filas].join("\n");
  const blob = new Blob([contenido], { type: "text/plain;charset=utf-8" });
  const url  = URL.createObjectURL(blob);
  const a    = document.createElement("a");
  a.href = url;
  a.download = `temperaturas_${Date.now()}.txt`;
  a.click();
  URL.revokeObjectURL(url);
}
